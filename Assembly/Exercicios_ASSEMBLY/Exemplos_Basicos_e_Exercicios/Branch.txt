;Leia um valor de PTAD
;Se PTA > 10, mostre em PTC:PTB <= 8*PTA    (parte alte a em PTC e baixa em PTB)
;   Senão, mostre em PTB =  PTA*25
;A*9 - 7

    ORG RAMStart
A   rmb 1 ;define uma var. de 1 byte

    ORG ROMStart
MAIN: clra              ;limpa o acumulador
      MOV #$FF, PTBDD   ;Porta B é saída
      MOV #$FF, PTCDD   ;Porta C é saída
      MOV #$00, PTADD   ;Porta A é entrada
      SEI               ;desliga interrupção
      
Loop: LDA PTAD        ;Acum <= Porta A
      SUB #10         ;Acum <= Acum - 10
      BGT Maior10     ;Se PTA>10 vai pro if.
      
Else: 
      LDA PTAD        ;Carrega denovo o valor da porta A
      LDX #25         ;Vai multiplicar com 25
      MUL
      STA PTBD        ;Salva parte baixa
      STX PTCD        ;Limpa o PTC , pois X vai ter 0.
      
      JMP Loop        ;Fica em loop
      
Maior10:
      LDA PTAD        ;Carrega denovo o valor da porta A
      LDX #8          ;Vai multiplicar com 8
      MUL
      STA PTBD        ;Salva parte baixa
      STX PTCD        ;Salva parte alta
      
      JMP Loop        ;Fica em loop
